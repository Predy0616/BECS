# 객체 지향 복습

## 다형성
### 동일한 이름의 함수가 다른 방식으로 동작하는 객체 지향 프로그래밍의 특징

### 예제


    #include<iostream>

    using namespace std;

    class Animal {
        public:
            virtual void sound() = 0; 
    }

    class Cat : public Animal {
        public:
            void sound() override{
                cout << "MEOW";
            }
    }

    class Dog : public Animal {
        public:
            void sound() override{
                cout << "BOW WOW";
            }
    }

    int main(){
        Animal* choco = new Dog();
        Animal* neko = new Cat();
        choco->sound();
        neko->sound();
        return 0;
    }
